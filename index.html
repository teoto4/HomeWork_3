function sumOfNumb(numb) {
numb = Math.abs(numb);

const arrNumber3 = [22, 45, 12];
function averaging(...arg) {
const sum = arg.reduce((accumulator, currentValue) => {
return accumulator + currentValue;
});
return sum / arg.length;
}
